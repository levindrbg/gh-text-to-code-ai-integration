This is an example script containing essential components and structure.import numpy as np# SECTION 1: INPUT PARAMETERS# --- TRUSS GEOMETRY & DOMAIN ---SPAN = 15000 # 15 meters in mmHEIGHT = 3000 # 3 meters in mmNUM_BAYS = 12# --- LOADING CONDITIONS ---FACTORED_LINE_LOAD_PER_MM = 750000 / (NUM_BAYS + 1) # Total load distributed across the bay width# --- MATERIAL PROPERTIES ---ALUMINUM_E = 70000.0 # Modulus of Elasticity for Aluminum in MPa (N/mm^2)CFRP_E = 230000.0 # Modulus of Elasticity for CFRP in MPa (N/mm^2)ALUMINUM_STRENGTH_TENSION = 200.0 # Allowable tensile stress in MPa for AluminumCFRP_STRENGTH_COMPRESSION = 300.0 # Allowable compressive stress in MPa for CFRPALUMINUM_DENSITY = 2.7e-6 # kg/mm^3 for AluminumCFRP_DENSITY = 1.6e-6 # kg/mm^3 for CFRP# --- OPTIMIZATION PARAMETERS ---INITIAL_AREA = 20000MIN_AREA = 1.0NUM_ITERATIONS = 120EVOLUTIONARY_RATE = 0.03MAX_STRESS_ADJUSTMENT_RATE = 0.5# --- BUCKLING PARAMETERS ---T_D_RATIO = 0.1K_FACTOR_BUCKLING = 1.0# SECTION 2: SETUP AND STRUCTURAL ANALYSISdef setup_truss_ground_structure(): """Creates the initial dense mesh of nodes and potential elements (ground structure).""" nodes = [] for i in range(NUM_BAYS + 1): x = i * (SPAN / NUM_BAYS) nodes.append([x, 0]) # Bottom chord node nodes.append([x, HEIGHT]) # Top chord node nodes = np.array(nodes) elements, chord_indices, web_indices = [], [], [] num_nodes_total = len(nodes) element_counter = 0 for i in range(num_nodes_total): for j in range(i + 1, num_nodes_total): node1_x, node1_y = nodes[i] node2_x, node2_y = nodes[j] if abs(node1_x - node2_x) <= 3 * (SPAN / NUM_BAYS): elements.append([i, j]) is_chord = (node1_y == node2_y) and (abs(node1_x - node2_x) <= (SPAN / NUM_BAYS) * 1.01) if is_chord: chord_indices.append(element_counter) else: web_indices.append(element_counter) element_counter += 1 elements = np.array(elements) bottom_right_node_idx = 2 * NUM_BAYS fixed_dofs = [ 0, 1, 2 * bottom_right_node_idx + 1, ] load_nodes = [1] load_values = np.array([[0, -FACTORED_LINE_LOAD_PER_MM]]) return nodes, elements, fixed_dofs, np.array(load_nodes), load_values, np.array(chord_indices), np.array(web_indices)def analyze_truss(nodes, elements, areas, fixed_dofs, external_loads): """Performs Finite Element Analysis (FEA) including self-weight.""" num_nodes = len(nodes) num_elements = len(elements) num_dofs = num_nodes * 2 K = np.zeros((num_dofs, num_dofs)) element_lengths = np.zeros(num_elements) self_weight_forces = np.zeros((num_nodes, 2)) for i, (n1_idx, n2_idx) in enumerate(elements): L = np.linalg.norm(nodes[n2_idx] - nodes[n1_idx]) element_lengths[i] = L member_weight = areas[i] * L * ALUMINUM_DENSITY * 9.81 / 1000.0 self_weight_forces[n1_idx, 1] -= member_weight / 2.0 self_weight_forces[n2_idx, 1] -= member_weight / 2.0 for i, (n1_idx, n2_idx) in enumerate(elements): L = element_lengths[i] if L == 0: continue c, s = (nodes[n2_idx, 0] - nodes[n1_idx, 0]) / L, (nodes[n2_idx, 1] - nodes[n1_idx, 1]) / L A = areas[i] k_e = (A * ALUMINUM_E / L) * np.array([ [c * c, c * s, -c * c, -c * s], [c * s, s * s, -c * s, -s * s], [-c * c, -c * s, c * c, c * s], [-c * s, -s * s, c * s, s * s] ]) dofs = [2 * n1_idx, 2 * n1_idx + 1, 2 * n2_idx, 2 * n2_idx + 1] K[np.ix_(dofs, dofs)] += k_e F = external_loads.flatten() + self_weight_forces.flatten() free_dofs = np.setdiff1d(range(num_dofs), fixed_dofs) U = np.zeros(num_dofs) try: U[free_dofs] = np.linalg.solve(K[np.ix_(free_dofs, free_dofs)], F[free_dofs]) except np.linalg.LinAlgError: return None, None, None, None strains = np.zeros(num_elements) for i, (n1_idx, n2_idx) in enumerate(elements): L = element_lengths[i] if L == 0: continue c, s = (nodes[n2_idx, 0] - nodes[n1_idx, 0]) / L, (nodes[n2_idx, 1] - nodes[n1_idx, 1]) / L u1x, u1y = U[2 * n1_idx], U[2 * n1_idx + 1] u2x, u2y = U[2 * n2_idx], U[2 * n2_idx + 1] strains[i] = (c * (u2x - u1x) + s * (u2y - u1y)) / L stresses = ALUMINUM_E * strains return U.reshape((num_nodes, 2)), strains, stresses, element_lengthsdef calculate_buckling_stress(areas, lengths): """Calculates the critical buckling stress for each member.""" safe_areas = np.maximum(areas, MIN_AREA) I = safe_areas ** 2 / (8 * np.pi * T_D_RATIO) effective_lengths = K_FACTOR_BUCKLING * lengths + 1e-9 critical_stress = (np.pi ** 2 * CFRP_E * I) / (effective_lengths ** 2 * safe_areas) return np.minimum(critical_stress, CFRP_STRENGTH_COMPRESSION)def update_areas_beso_with_chords(areas, stresses, element_lengths, chord_indices, web_indices): """Updates areas using BESO, with materials considered for buckling.""" new_areas = areas.copy() allowable_compressive_stress = calculate_buckling_stress(areas, element_lengths) allowable_stresses = np.where(stresses >= 0, ALUMINUM_STRENGTH_TENSION, allowable_compressive_stress) allowable_stresses[allowable_stresses < 1e-3] = 1.0 stress_ratio = np.abs(stresses) / allowable_stresses target_areas = areas * stress_ratio new_areas = (1 - MAX_STRESS_ADJUSTMENT_RATE) * areas + MAX_STRESS_ADJUSTMENT_RATE * target_areas return np.maximum(MIN_AREA, new_areas)def export_final_design_to_json(nodes, elements, areas): """Exports the final truss design data as a list of dictionaries.""" results = [] for i, (n1_idx, n2_idx) in enumerate(elements): if areas[i] <= MIN_AREA * 1.1: continue data = { "member_id": f"member_{i + 1:02d}", "start_x": nodes[n1_idx][0] / 1000.0, "start_y": nodes[n1_idx][1] / 1000.0, "end_x": nodes[n2_idx][0] / 1000.0, "end_y": nodes[n2_idx][1] / 1000.0, "area": areas[i] / 100.0 } results.append(data) return results# SECTION 5: MAIN EXECUTION SCRIPTdef generate_truss_design(): # Setup truss structure nodes, elements, fixed_dofs, load_nodes, load_values, chord_indices, web_indices = setup_truss_ground_structure() num_elements = len(elements) areas = np.full(num_elements, INITIAL_AREA, dtype=float) # Apply loads external_loads = np.zeros((len(nodes), 2)) for i, node_idx in enumerate(load_nodes): external_loads[node_idx, :] += load_values[i, :] # Run optimization for i in range(NUM_ITERATIONS): analysis_results = analyze_truss(nodes, elements, areas, fixed_dofs, external_loads) if analysis_results[0] is None: break displacements, strains, stresses, element_lengths = analysis_results areas = update_areas_beso_with_chords(areas, stresses, element_lengths, chord_indices, web_indices) # Prepare final output final_results = export_final_design_to_json(nodes, elements, areas) return final_results # Explicit return of the array of dictionaries# This is the output that n8n will receiveoutput_data = generate_truss_design()return output_data
