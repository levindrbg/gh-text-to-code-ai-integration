5. format_results_as_jsonpythondef format_results_as_json(nodes, elements, areas, stresses, element_lengths): """Formats results as JSON-serializable dictionary. Returns: dict: Optimized truss data ready for JSON serialization """ active_mask = areas > MIN_AREA * 1.1 total_volume_m3 = np.sum(areas[active_mask] * element_lengths[active_mask]) / 1e9 total_mass_kg = total_volume_m3 * (STEEL_DENSITY * 1e9) member_data = [] for i, (n1_idx, n2_idx) in enumerate(elements): if not active_mask[i]: continue stress = stresses[i] force_kN = (stress * areas[i]) / 1000.0 status = "Tension" if stress >= 0 else "Compression" start_node = {'x': round(nodes[n1_idx, 0]/1000, 3), 'y': round(nodes[n1_idx, 1]/1000, 3)} end_node = {'x': round(nodes[n2_idx, 0]/1000, 3), 'y': round(nodes[n2_idx, 1]/1000, 3)} member_data.append({ 'memberId': i, 'startNode': start_node, 'endNode': end_node, 'length_m': round(element_lengths[i]/1000.0, 3), 'area_cm2': round(areas[i]/100.0, 2), 'stress_MPa': round(stress, 2), 'force_kN': round(force_kN, 2), 'status': status }) return { "summary": { "totalActiveMembers": len(member_data), "totalVolume_m3": round(total_volume_m3, 4), "totalMass_kg": round(total_mass_kg, 2) }, "members": member_data }Metadata:Dependencies: numpy, json, Globals: MIN_AREA, STEEL_DENSITYInputs:nodes: Node coordinateselements: Member connectionsareas: Final member areasstresses: Final member stresseselement_lengths: Member lengthsOutput: Dictionary containing truss summary and member detailsDescription:Prepares final results for output by:Filtering active members (>1.1×MIN_AREA)Calculating total volume and massConverting units (mm → m, mm² → cm², N → kN)Packaging member data with coordinatesStructuring JSON-ready output
