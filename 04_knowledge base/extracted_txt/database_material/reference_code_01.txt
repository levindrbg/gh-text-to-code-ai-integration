setup_truss_ground_structure(): nodes = [] for i in range(NUM_BAYS + 1): x = i * (SPAN / NUM_BAYS) nodes.append([x, 0]) nodes.append([x, HEIGHT]) nodes = np.array(nodes) elements = [] num_nodes_total = len(nodes) for i in range(num_nodes_total): for j in range(i + 1, num_nodes_total): if abs(nodes[i][0] - nodes[j][0]) <= 2 * (SPAN / NUM_BAYS): elements.append([i, j]) elements = np.array(elements) bottom_right_node_idx = 2 * NUM_BAYS fixed_dofs = [0, 1, 2 * bottom_right_node_idx + 1] middle_node_index = NUM_BAYS + 1 load_nodes = [middle_node_index] load_values = np.array([[0, -POINT_LOAD]]) return nodes, elements, fixed_dofs, load_nodes, load_values# --- FEA FUNCTION ---def analyze_truss(nodes, elements, areas, materials, fixed_dofs, load_nodes, load_values): num_nodes = len(nodes) num_elements = len(elements) num_dofs = num_nodes * 2 K = np.zeros((num_dofs, num_dofs)) element_lengths = np.zeros(num_elements) for i, (n1_idx, n2_idx) in enumerate(elements): n1, n2 = nodes[n1_idx], nodes[n2_idx] L = np.linalg.norm(n2 - n1) if L == 0: continue element_lengths[i] = L c = (n2[0] - n1[0]) / L s = (n2[1] - n1[1]) / L cc, cs, ss = c * c, c * s, s * s E = materials[i]['E'] A = areas[i] k_e = (A * E / L) * np.array([ [cc, cs, -cc, -cs], [cs, ss, -cs, -ss], [-cc, -cs, cc, cs], [-cs, -ss, cs, ss] ]) dofs = [2 * n1_idx, 2 * n1_idx + 1, 2 * n2_idx, 2 * n2_idx + 1] for row, dof_r in enumerate(dofs): for col, dof_c in enumerate(dofs): if dof_r < num_dofs and dof_c < num_dofs: K[dof_r, dof_c] += k_e[row, col] F = np.zeros(num_dofs) for i, node_idx in enumerate(load_nodes): dof_x, dof_y = 2 * node_idx, 2 * node_idx + 1 F[dof_x] += load_values[i, 0] F[dof_y] += load_values[i, 1] free_dofs = np.setdiff1d(np.arange(num_dofs), fixed_dofs) U = np.zeros(num_dofs) try: if len(free_dofs) > 0: U[free_dofs] = np.linalg.solve(K[np.ix_(free_dofs, free_dofs)], F[free_dofs]) except np.linalg.LinAlgError: print("Singular matrix — structure unstable.") return None, None, None, None strains = np.zeros(num_elements) for i, (n1_idx, n2_idx) in enumerate(elements): L = element_lengths[i] if L == 0: continue n1, n2 = nodes[n1_idx], nodes[n2_idx] c = (n2[0] - n1[0]) / L s = (n2[1] - n1[1]) / L u1x, u1y = U[2 * n1_idx], U[2 * n1_idx + 1] u2x, u2y = U[2 * n2_idx], U[2 * n2_idx + 1] strains[i] = (1 / L) * (c * (u2x - u1x) + s * (u2y - u1y)) stresses = np.array([materials[i]['E'] * strains[i] for i in range(num_elements)]) return U.reshape((num_nodes, 2)), strains, stresses, element_lengths# --- VISUALIZATION ---def plot_truss(ax,
