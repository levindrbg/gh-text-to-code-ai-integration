5. format_results_as_jsondef format_results_as_json(nodes, elements, areas, stresses=None, element_lengths=None): """Formats truss member results in a specific JSON structure with each member wrapped in a "json" object. Args: nodes: Node coordinates in mm [Nx2 or Nx3 array] elements: Member connections as node index pairs [Mx2 array] areas: Cross-sectional areas in mm² [M array] stresses: (Optional) Member stresses in MPa [M array] element_lengths: (Optional) Member lengths in mm [M array] (not currently used) Returns: list: List of dictionaries where each contains a "json" key with member data Example Output: [ {"json": {"member_id": "member_01", "start_x": 0.0, "start_y": 0.0, "start_z": 0.0, "end_x": 2.0, "end_y": 0.0, "end_z": 0.0, "area": 12.34}}, {"json": {"member_id": "member_02", "start_x": 2.0, "start_y": 0.0, "start_z": 0.0, "end_x": 4.0, "end_y": 0.0, "end_z": 0.0, "area": 10.50, "stress_MPa": 125.67}}, ... ] Description: - Converts units from mm to m for coordinates (3 decimal places) - Converts areas from mm² to cm² (2 decimal places) - Handles both 2D (z=0) and 3D cases automatically - Uses 1-based numbering for member IDs (member_01 to member_NN) - Formats areas as empty string when below threshold - Includes stress data when provided - Wraps each member in a {"json": {...}} structure - Filters out members with areas below MIN_AREA threshold """ MIN_AREA = 1e-6 # Minimum area threshold in mm² active_mask = areas > MIN_AREA * 1.1 member_list = [] for i, (n1_idx, n2_idx) in enumerate(elements): if not active_mask[i]: continue member_data = { "member_id": f"member_{i+1:02d}", # 1-based indexing "start_x": round(nodes[n1_idx, 0]/1000, 3), # mm to m "start_y": round(nodes[n1_idx, 1]/1000, 3), "start_z": round(nodes[n1_idx, 2]/1000, 3) if nodes.shape[1] > 2 else 0.0, "end_x": round(nodes[n2_idx, 0]/1000, 3), "end_y": round(nodes[n2_idx, 1]/1000, 3), "end_z": round(nodes[n2_idx, 2]/1000, 3) if nodes.shape[1] > 2 else 0.0, "area": round(areas[i]/100.0, 2) if areas[i] > MIN_AREA else "" # mm² to cm² } # Optional stress data if stresses is not None and i < len(stresses): member_data["stress_MPa"] = round(stresses[i], 2) member_list.append({"json": member_data}) return member_list
